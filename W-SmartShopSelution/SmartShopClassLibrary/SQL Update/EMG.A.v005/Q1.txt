
// Add the OrderProduct update

CREATE PROCEDURE spOrderProduct_Update
	
	@Id int,
	@Quantity float

AS
BEGIN


	SET NOCOUNT ON;

	UPDATE OrderProduct
	SET Quantity = @Quantity
	WHERE Id = Id;
	
    
END
GO

// Removing the orderpayment From the database

CREATE PROCEDURE spOrderPayment_Delete
	@Id int
AS
BEGIN
	

	SET NOCOUNT ON;

	DELETE FROM OrderPayment
	WHERE Id = @Id;
    
END
GO
